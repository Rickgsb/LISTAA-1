namespace PJEX06
{
    internal class Converter
    {
        //atributos
        private double mm;
        private double km;
        

        //metodos 
        public void setMilhas(double p)
        {
            mm = p;
        }


        public double getMilhas()
        {
            return mm;
        }

        public double getQuilometros()
        {
            return km;
        }


        public void calcularQuilometros()
        {
            km =(mm*1.852);
        }
    }
}
