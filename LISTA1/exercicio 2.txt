namespace EX2
{
    internal class Quadrado
    {
       
        private int l1;
        private int a;

    
        public void setAresta(int p)
        {
            l1 = p;
        }

        public int getAresta()
        {
            return l1;
        }

        public int getArea()
        {
            return a;
        }

        public void calcularArea()
        {
            a = l1 * l1;
        }
    }
}
